// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table users {
  id int
  full_name string
  email varchar(unique)
  password varchar
  role enum('guest', 'user')
  created_at timestamp
}


Table plans {
  id int
  name enum('premium', 'pro')
  price decimal
  duration_days enum
}

Table plan_features{
  id int
  plan_id int [ref: > plans.id]
  max_device int
  max_scan_per_day int
}

Table user_subscriptions{
  id int
  plan_features_id int [ref: > plan_features.id]
  user_id int [ref: > users.id]
  is_active bool
  start_at timestamp
  expired_at timestamp
  }

Table payments{
  id int
  user_subscription_id int [ref: > user_subscriptions.id]
  payment_status_id int [ref: > payment_status.id]
  external_id int
  amount_to_pay decimal
  paid_at timestamp
  created_at timestamp
}

Table payment_status{
  id int
  status enum('pending', 'success', 'failed', 'cancelled')
}

Table devices{
  id int
  hwid varchar(unique)
  user_id int [ref: > users.id]
  os_platform enum('Linux', 'Windows', 'Android', 'macOS', 'IOS')
  app_platform enum('Desktop', 'Web_app', 'Mobile')
  hostname varchar
  created_at timestamp
  last_active timestamp
}

Table scan_jobs{
  id int
  user_id int [ref: > users.id]
  device_id int [ref: > devices.id]
  scan_types_id int [ref: > scan_types.id]
  status enum('pending', 'processing', 'completed', 'failed', 'cancelled')
  total_files_scanned int
  malware_found int
  submitted_at timestamp
  completed_at timestamp
}

Table scan_types{
  id int
  name enum('full', 'quick')
}

Table scan_results{
  id int
  scan_job_id int [ref: > scan_jobs.id]
  file_name varchar
  file_path varchar
  prediction enum('malware', 'benign')
}

Table scan_usage{
  id int 
  user_id int [ref: > users.id]
  scan_types_id int [ref: > scan_types.id]
  date date
  total_scan int
}

Table model_version{
  id int
  version_name varchar
  accuracy float
  model_path string
  created_at timestamp
  is_active bool
}
